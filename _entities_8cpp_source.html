39  {
40  // SAFETY: firstUnset works because there is always a sentinel zero bit at the end of the 'alive' set.
41  std::size_t firstDead = this->_alive.firstUnset();
42 
43  if (firstDead == this->_generations.size()) {
44  // If there is not empty slot (apart from the sentinel), add one to the end

46  this->_alive[firstDead] = alive;
