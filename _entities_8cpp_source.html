34  Entity Entities::create(bool alive)
35  {
36  // SAFETY: firstUnset works because there is always a sentinel zero bit at the end of the 'alive' set.
37  std::size_t firstDead = this->_alive.firstUnset();
38 
39  if (firstDead == this->_generations.size()) {
40  // If there is not empty slot (apart from the sentinel), add one to the end
41  this->_generations.push_back(1);
42  this->_alive[firstDead] = alive;
