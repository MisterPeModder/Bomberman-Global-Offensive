39  {

41  this->_timeSinceRepeat = std::min(100.0, this->_timeSinceRepeat + elapsed);

127  void KeyboardInput::moveCursor(int offset, bool selectingText)
128  {
129  if (this->contents.empty())
130  return;
131 
132  Vector2u pos = selectingText ? this->selectionPos : this->cursorPos;
133 
134  // normalize cursor height
135  if (pos.y >= this->contents.size())
136  pos.y = this->contents.size() - 1;
137 
138  if (offset < 0) {
139  if (pos.x == 0 && pos.y > 0) {
140  pos = Vector2u(this->contents[pos.y - 1].size(), pos.y - 1);
141  } else if (pos.x != 0) {
142  pos.x -= std::min(pos.x, static_cast<unsigned int>(-offset));
143  pos.x = util::utf8BoundaryStart(this->contents[pos.y], pos.x);
144  }
145  } else {
146  if (pos.x >= this->contents[pos.y].size() && pos.y + 1 < this->contents.size()) {
147  pos = Vector2u(0, pos.y + 1);
148  } else {
149  pos.x = std::min(this->contents[pos.y].size(), pos.x + static_cast<std::size_t>(offset));
150  pos.x = util::utf8BoundaryEnd(this->contents[pos.y], pos.x);
151  }
152  }
153 
154  this->setCursorPos(pos, selectingText);
155  }

157  void KeyboardInput::moveCursorToWord(int wordOffset, bool selectingText)
158  {
159  Vector2u pos = selectingText ? this->selectionPos : this->cursorPos;
160  int offset = 0;
161 
162  if (!this->contents.empty() && !this->contents[pos.y].empty()) {
163  std::string const &line = this->contents[pos.y];
164 
165  if (wordOffset > 0) {
166  while (pos.x + offset < line.size() && std::isalnum(line[pos.x + offset]))
167  ++offset;
168  while (pos.x + offset < line.size() && !std::isalnum(line[pos.x + offset]))
169  ++offset;
170  } else if (wordOffset < 0) {
171  while (pos.x - offset > 0 && std::isalnum(line[pos.x - offset - 1]))
172  ++offset;
173  while (pos.x - offset > 0 && !std::isalnum(line[pos.x - offset - 1]))
174  ++offset;
175  offset = -offset;
176  }
177  }
178  this->moveCursor(offset, selectingText);

205  void KeyboardInput::eraseSelection()
206  {
207  auto [selectStart, selectEnd] = this->getSelection();
208  unsigned int lineCount = selectEnd.y - selectStart.y;
209  std::string &firstLine = this->contents[selectStart.y];
210 
211  this->cursorPos = selectStart;
212  this->selectionPos = selectStart;
213  if (lineCount == 0) {
214  firstLine.erase(firstLine.cbegin() + selectStart.x, firstLine.cbegin() + selectEnd.x);
215  return;
216  }
217  std::string &lastLine = this->contents[selectEnd.y];
218 
219  firstLine.erase(firstLine.cbegin() + selectStart.x, firstLine.cend());
220 
221  if (selectEnd.x >= lastLine.size())
222  this->contents.erase(this->contents.cbegin() + selectEnd.y);
223  else
224  lastLine.erase(lastLine.cbegin(), lastLine.cbegin() + selectEnd.x);
225 
226  if (lineCount > 1)
227  this->contents.erase(this->contents.cbegin() + selectStart.y + 1, this->contents.cbegin() + selectEnd.y);

310  {

312 
313  if (this->hasSelection())
314  this->eraseSelection();
315 
316  auto end = pasted.cend();

318 
319  std::size_t startY = this->cursorPos.y;
320 
321  this->contents[startY].insert(this->contents[startY].cbegin() + this->cursorPos.x, pasted.cbegin(), it);
322 
323  size_t firstPasteLineSize = it - pasted.cbegin();
324  size_t firstPasteLineEnd = this->cursorPos.x + firstPasteLineSize;
325  this->setCursorPos({static_cast<unsigned int>(firstPasteLineEnd), cursorPos.y});
326 
327  for (std::size_t lineY = this->cursorPos.y; it != end; ++lineY) {
328  auto next = std::find(++it, end, '\n');
329  if (next != end) {
330  this->contents.emplace(this->contents.cbegin() + lineY + 1, it, next);
331  } else {
332  if (lineY + 1 >= this->contents.size())
333  this->contents.emplace_back();

335 
336  size_t lastPasteLineSize = next - it;
337 
338  this->setCursorPos(
339  {static_cast<unsigned int>(lastPasteLineSize), static_cast<unsigned int>(lineY + 1)});
340 
341  auto rest = this->contents[startY].cbegin() + firstPasteLineEnd;
342  this->contents[lineY + 1].insert(

397  void KeyboardInput::deleteForwards()
398  {
399  if (this->hasSelection()) {
400  this->eraseSelection();
401  return;
402  }
403  Vector2u pos = this->cursorPos;
404 
405  // do nothing when ...
406  if (this->contents.empty() // ...field has no contents...
407  || pos.y >= this->contents.size() // ...cursor Y is beyond the number of lines...
408  || (pos.y + 1 == this->contents.size()
409  && pos.x >= this->contents[pos.y].size())) // ...cursor is past the last char.
410  return;
411 
412  std::string const &currentLine = this->contents[pos.y];
413 
414  if (pos.y + 1 < this->contents.size() && pos.x >= currentLine.size()) {
415  this->moveCursor(1);
416  this->contents[pos.y + 1].append(currentLine);
417  this->contents.erase(this->contents.begin() + pos.y);
418  } else {
419  util::removeUtf8Codepoint(this->contents[pos.y], pos.x);
420  this->moveCursor(0);
421  }
