62  struct DetectGamepad : public ecs::System {
63  void run(ecs::SystemData data) override final
64  {
65  auto engine = data.getResource<game::resources::EngineResource>().engine;
66  auto &users = engine->getUsers();
67 
68  int gamepadId = users.getJoiningGamepad();
69  if (gamepadId == -1)
70  return;
71  raylib::core::Gamepad::Button pressedBtn = raylib::core::Gamepad::getButtonPressed();
72 
73  if (pressedBtn == raylib::core::Gamepad::Button::MIDDLE_LEFT)
74  return;
75  if (pressedBtn != raylib::core::Gamepad::Button::MIDDLE_RIGHT
76  && users[game::User::UserId::User1].isKeyboard()) {
77  users[game::User::UserId::User1].setGamepadId(gamepadId);
78  Logger::logger.log(Logger::Severity::Information, "User 1 switched to gamepad mode.");
79  } else {
80  users.connectUser(gamepadId, dynamic_cast<game::MainMenuScene &>(engine->getScene()).getUnusedSkin());
81  dynamic_cast<game::MainMenuScene &>(engine->getScene()).updateConnectedTexts();
82  dynamic_cast<game::MainMenuScene &>(engine->getScene()).updateSkins();
83  }
84  }
85  };

252  if (id == 0)
253  _world.getStorage<components::Model>()[playerEntity.getId()].setMaterialMapTexture(
254  textures.get(std::string(userSkinToRessourceString(_availableSkins.front()).getMsgId())));
255  else

257  textures.get(std::string(localization::resources::textures::rsUnknown.getMsgId())));

259  auto &anim = _world.getStorage<components::Animation>()[playerEntity.getId()];
260  auto &randDevice = _world.getResource<game::resources::RandomDevice>();
261  unsigned int randVal = randDevice.randInt(0, 3);

263  _models[id] = _world.getStorage<components::Identity>()[playerEntity.getId()].id;
264  anim.chooseAnimation(randVal);
265 
266  // Skin Text
267  auto builder = _world.addEntity();
268  (void)builder.with<components::Position>(17 + static_cast<int>(id) * 19 + static_cast<int>(id) / 2, 75)
269  .with<components::Identity>()
270  .with<components::Controlable>(User::UserId::AllUsers,
271  [this](ecs::Entity controlable, ecs::SystemData data, const Users::ActionEvent &event) {
272  (void)controlable;

278  if (event.action == GameAction::NEXT_ACTIVABLE) {
279  _availableSkins.push_back(user.getSkin());
280  user.setSkin(_availableSkins.front());
281  _availableSkins.pop_front();
282  } else {
283  _availableSkins.push_front(user.getSkin());
284  user.setSkin(_availableSkins.back());
285  _availableSkins.pop_back();
286  }
287  updateSkins();
288  return true;
289  });
290  if (id == 0) {
291  (void)builder.with<components::Textual>(
292  userSkinToRessourceString(_availableSkins.front()), 20, raylib::core::Color::WHITE);
293  _availableSkins.pop_front();
294  } else {
295  (void)builder.with<components::Textual>(
296  userSkinToRessourceString(User::USER_SKINS::UNKNOWN), 20, raylib::core::Color::WHITE);
297  }
298  auto skinText = builder.build();

300

370 
371  // update Text

374  for (size_t i = 0; i < static_cast<size_t>(User::UserId::UserCount); i++) {

377  User::UserId userId = static_cast<User::UserId>(i);

379  }
380  }
381  // Update texture

383  ecs::join(_world.getStorage<components::Model>(), _world.getStorage<components::Identity>())) {
384  for (size_t i = 0; i < static_cast<size_t>(User::UserId::UserCount); i++) {

386  continue;
387  User::UserId userId = static_cast<User::UserId>(i);
388  model.setMaterialMapTexture(
389  textures.get(std::string(userSkinToRessourceString(users[userId].getSkin()).getMsgId())));
390  }
