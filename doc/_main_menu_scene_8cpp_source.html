54  struct DetectGamepad : public ecs::System {
55  void run(ecs::SystemData data) override final
56  {
57  auto engine = data.getResource<game::resources::EngineResource>().engine;
58  auto &users = engine->getUsers();
59 
60  int gamepadId = users.getJoiningGamepad();
61  if (gamepadId == -1)
62  return;
63  raylib::core::Gamepad::Button pressedBtn = raylib::core::Gamepad::getButtonPressed();
64 
65  if (pressedBtn == raylib::core::Gamepad::Button::MIDDLE_LEFT)
66  return;
67  if (pressedBtn != raylib::core::Gamepad::Button::MIDDLE_RIGHT
68  && users[game::User::UserId::User1].isKeyboard()) {
69  users[game::User::UserId::User1].setGamepadId(gamepadId);
70  Logger::logger.log(Logger::Severity::Information, "User 1 switched to gamepad mode.");
71  } else {
72  users.connectUser(gamepadId);
73  dynamic_cast<game::MainMenuScene &>(engine->getScene()).updateConnectedTexts();
74  }
75  }
76  };

206  void MainMenuScene::updateConnectedTexts()
207  {
208  auto engine = _world.getResource<game::resources::EngineResource>().engine;
209  auto &users = engine->getUsers();
210 
211  for (auto [text, id] :
212  ecs::join(_world.getStorage<components::Textual>(), _world.getStorage<components::Identity>())) {
213  if (id.id >= _firstUserId && id.id < _firstUserId + 4) {
214  User::UserId userId = static_cast<User::UserId>(id.id - _firstUserId);
215 
216  if (users[userId].isAvailable()) {
217  text.text = localization::resources::menu::rsConnected;
218  text.color = raylib::core::Color::GREEN;
219  } else {
220  text.text = localization::resources::menu::rsNotConnected;

222  }
