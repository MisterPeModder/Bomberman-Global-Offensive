121  bool Player::handleActionEvent(ecs::Entity self, ecs::SystemData data, const Users::ActionEvent &event)
122  {
123  if (data.getResource<game::resources::GameClock>().isPaused() && event.action != GameAction::PAUSE)
124  return false;
125  if (isMoveAction(event.action))
126  move(self, data, event);
127  else if (event.action == GameAction::PLACE_BOMB && event.value > 0.9f)
128  placeBomb(self, data);
129  else if (event.action == GameAction::ACTIVATE_ITEM && event.value > 0.9f)
130  data.getStorage<Player>()[self.getId()].inventory.useActivable(self, data);
131  else if (event.action == GameAction::PREVIOUS_ACTIVABLE && event.value > 0.9f)
132  data.getStorage<Player>()[self.getId()].inventory.selectPreviousActivable();
133  else if (event.action == GameAction::NEXT_ACTIVABLE && event.value > 0.9f)
134  data.getStorage<Player>()[self.getId()].inventory.selectNextActivable();
135  else if (event.action == GameAction::PAUSE && event.value > 0.9f)
136  data.getResource<game::resources::GameClock>().togglePause();
137  else
138  return false;
139  return true;

142  void Player::move(ecs::Entity self, ecs::SystemData data, GameAction action, float value)
143  {
144  auto &velocity = data.getStorage<Velocity>()[self.getId()];
145  auto &rAngle = data.getStorage<RotationAngle>()[self.getId()];
146  auto &anim = data.getStorage<Animation>()[self.getId()];
147  auto &stats = data.getStorage<Player>()[self.getId()].stats;
148  auto &player = data.getStorage<Player>()[self.getId()];
149 
150  if (data.getResource<game::resources::GameClock>().getTime() < player.stats.stunEnd)
151  return;
152 
153  if (value < 0.2f) {
154  velocity = {0.f, 0.f, 0.f};
155  if (player.animation == Animations::Run) {
156  // Randomize the idle animation id
157  auto &randDevice = data.getResource<game::resources::RandomDevice>();
158  unsigned int randVal = randDevice.randInt(
159  static_cast<unsigned int>(Animations::Idle_1), static_cast<unsigned int>(Animations::Idle_4));
160 
161  anim.chooseAnimation(randVal);
162  player.animation = Animations(randVal);
163  }
164  } else {
165  float speed = stats.speed;
166 
167  if (stats.inverted)
168  speed *= -1.f;
169  if (stats.slowness)
170  speed *= 0.25f;
171  switch (action) {
172  case GameAction::MOVE_LEFT:
173  velocity = {-speed, 0.f, 0.f};
174  rAngle.rotationAngle = 270.f;
175  break;
176  case GameAction::MOVE_UP:
177  velocity = {0.f, 0.f, -speed};
178  rAngle.rotationAngle = 180.f;
179  break;
180  case GameAction::MOVE_RIGHT:
181  velocity = {speed, 0.f, 0.f};
182  rAngle.rotationAngle = 90.f;
183  break;
184  case GameAction::MOVE_DOWN:
185  velocity = {0.f, 0.f, speed};
186  rAngle.rotationAngle = 0.f;
187  break;
188  default: break;
189  }
190  if (velocity != raylib::core::Vector3f(0.f, 0.f, 0.f) && player.animation != Animations::Run) {
191  anim.chooseAnimation(static_cast<unsigned int>(Animations::Run));
192  player.animation = Animations::Run;
193  }
194  }
