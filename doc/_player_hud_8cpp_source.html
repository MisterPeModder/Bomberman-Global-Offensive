163  void PlayerHud::update(ecs::SystemData data, const Player &player)
164  {

166  if (id.id == getInfoId(Info::C4_Count)) {
167  size_t bombs = player.stats.bombLimit;
168  text.text = std::to_string(bombs - player.placedBombs) + "/" + std::to_string(bombs);
169  } else if (id.id == getInfoId(Info::Speed))
170  text.text = std::to_string(player.stats.speed).substr(0, 3);
171  else if (id.id == getInfoId(Info::Range))
172  text.text = std::to_string(player.stats.bombRange);
173  else {
174  size_t firstItem = static_cast<size_t>(Item::FIRST_ACTIVABLE);
175  for (size_t i = firstItem; i < static_cast<size_t>(Item::Identifier::Count); i++)
176  if (id.id == getInfoId(static_cast<Info>(static_cast<size_t>(Info::NoClip) + i - firstItem))) {
177  size_t count = player.inventory[static_cast<Item::Identifier>(i)];
178  text.text = std::to_string(count);
179  text.color = (count > 0) ? raylib::core::Color::GREEN : raylib::core::Color::RED;
180  }
181  }
182  }
183 
184  for (auto [pos, id] : ecs::join(data.getStorage<Position>(), data.getStorage<Identity>())) {
185  if (id.id == getInfoId(Info::SelectedItem))
