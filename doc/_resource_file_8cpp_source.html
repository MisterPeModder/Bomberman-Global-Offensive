190  void ResourceFile::readMsg(
191  bool multiline, std::string &out, const TokensVector &tokens, TokensVector::const_iterator &iterator)
192  {
193  static std::regex singleLine("msg.+[ \t]+\"(.+)\"");
194  static std::regex msgMultipleLines("\"(.+)(\\\\n)\"");
195  std::smatch sm;
196 
197  if (multiline) {
198  skipComments(tokens, ++iterator);
199  while (iterator->first == Token::MsgMultiline) {
200  std::regex_match(iterator->second, sm, msgMultipleLines);
201  out += sm[1].str() + "\n";
202  skipComments(tokens, ++iterator);
203  }
204  if (out.size() > 0 && out[out.size() - 1] == '\n')
205  out.pop_back();
206  } else {
207  std::regex_match(iterator->second, sm, singleLine);
208  out = sm[1].str();
209  skipComments(tokens, ++iterator);
210  }
211  }

232  ResourceFile::Token ResourceFile::getToken(const std::string &line)
233  {
234  static std::regex comment("[ \t]*#.*");
235  static std::regex msgId("msgid[ \t]+\".+\"");
236  static std::regex msgIdEmpty("msgid[ \t]+\"\"");
237  static std::regex msgStr("msgstr[ \t]+\".+\"");
238  static std::regex msgStrEmpty("msgstr[ \t]+\"\"");
239  static std::regex msgMultipleLines("^[ \t]*\".+\\\\n\"");
240 
241  if (std::regex_match(line, comment))
242  return Token::Comment;
243  if (std::regex_match(line, msgId))
244  return Token::MsgId;
245  if (std::regex_match(line, msgIdEmpty))
246  return Token::MsgIdEmpty;
247  if (std::regex_match(line, msgStr))
248  return Token::MsgStr;
249  if (std::regex_match(line, msgStrEmpty))
250  return Token::MsgStrEmpty;
251  if (std::regex_match(line, msgMultipleLines))
252  return Token::MsgMultiline;
253  if (line == "")
254  return Token::EmptyLine;
255  return Token::Undefined;
256  }
