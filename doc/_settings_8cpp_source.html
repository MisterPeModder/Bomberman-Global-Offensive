52  void Settings::loadValue(std::string_view key, std::string_view value)
53  {
54  if (key == "sfx_volume")
55  setSfxVolume(std::stof(value.data()));
56  else if (key == "music_volume")
57  setMusicVolume(std::stof(value.data()));
58  else if (key == "master_volume")
59  setMasterVolume(std::stof(value.data()));
60  else if (key == "target_framerate")
61  setTargetFramerate(std::stoul(value.data()));
62  else if (key == "resolution") {
63  int width;
64  int height;
65 
66  if (value.find('x') == std::string::npos)
67  throw InvalidSettingsValue("Resolution must be of form '[width]x[height]'");
68  width = std::stoi(value.data());
69  height = std::stoi(value.data() + value.find('x') + 1);
70  setResolution({width, height});
71  } else if (key == "fullscreen") {
72  bool fullscreen;
73  std::istringstream(value.data()) >> fullscreen;
74  setFullscreen(fullscreen || value == "true");
75  } else if (key == "locale") {
76  setLocale(value);
77  } else {
78  Logger::logger.log(Logger::Severity::Warning, [&](std::ostream &writer) {
79  writer << "Unknown settings key '" << key << "' with value '" << value << "'";
80  });
81  }
