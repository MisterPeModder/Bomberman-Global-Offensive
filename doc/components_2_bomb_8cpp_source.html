36  void Bomb::explode(const Position &bombPos, ecs::SystemData data, ecs::Entity self)
37  {
39  if (this->exploded)
40  return;
41  this->exploded = true;
42  Sound::playSound(data, "C4");
43 
44  if (type == Type::Classic) {
45  for (auto [id, player] : ecs::join(data.getStorage<Identity>(), data.getStorage<Player>())) {
46  if (id.id == owner)
47  --player.placedBombs;
48  }

51  auto &positions = data.getStorage<Position>();
52  auto &sizes = data.getStorage<Size>();
53  auto maybeDestructible = ecs::maybe(data.getStorage<Destructible>());
54  auto maybeLiving = ecs::maybe(data.getStorage<Living>());
55  auto maybeBomb = ecs::maybe(data.getStorage<Bomb>());
56  auto &entities = data.getResource<ecs::Entities>();
57 
58  entities.kill(self);
60  game::map::Map &map = data.getResource<game::resources::Map>().map;
61  std::vector<raylib::core::Vector2u> explodedPositions;
62 
63  map.fillExplodedPositions(explodedPositions,
64  raylib::core::Vector2u(static_cast<unsigned int>(bombPos.x), static_cast<unsigned int>(bombPos.z)),
65  this->radius);
66  if (explodedPositions.empty())
67  return;
68  for (auto iter : explodedPositions)
69  Explosion::placeExplosion(data, iter);
70 
72  for (auto [pos, size, destructible, living, bomb, id] :
73  ecs::join(positions, sizes, maybeDestructible, maybeLiving, maybeBomb, entities)) {
75  if ((!destructible && !living && !bomb) || (living && living->hp == 0))
76  continue;
77 
78  raylib::core::Vector2u roundedPos2D = game::Game::worldPosToMapCell(pos);
79 
80  if (std::find(explodedPositions.begin(), explodedPositions.end(), roundedPos2D) == explodedPositions.end()
81  && (!living || !explodeLiving(pos, size, roundedPos2D, explodedPositions)))
82  continue;
83 
84  if (living) {
85  living->hp--;
86  if (living->hp == 0)
87  entities.kill(id);
88  }
89  if (destructible) {
90  entities.kill(id);
91  auto &elem = map.getElement(roundedPos2D);
92  if (elem == game::map::Map::Element::Crate)
93  Item::spawnRandomItem(data, roundedPos2D);
94  elem = game::map::Map::Element::Empty;
95  }
96  if (bomb)
97  bomb->explode(pos, data, id);
98  }
99  }
