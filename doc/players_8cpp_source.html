35  static std::optional<T> withPlayer(
36  bmjs::Number playerId, std::shared_ptr<bmjs::Engine> &engine, std::function<T(ecs::Entity, ecs::World &)> func)
37  {
38  int intId = static_cast<int>(playerId);
39 
40  if (!engine || intId < 0 || intId >= static_cast<int>(game::User::UserId::UserCount))
41  return {};
42  auto id = static_cast<game::User::UserId>(intId);
43 
44  auto &world = engine->getGameEngine().getScene().getWorld();
45  auto &players = world.getStorage<game::components::Player>();
46  auto &controlables = world.getStorage<game::components::Controlable>();
47  auto &entities = world.getResource<ecs::Entities>();
48 
49  auto join = ecs::join(players, controlables, entities);
50 
51  auto found = std::find_if(join.begin(), join.end(), [id](auto i) { return std::get<1>(i).userId == id; });
52 
53  if (found == join.end())
54  return {};
55  return func(std::get<2>(*found), world);
56  }
